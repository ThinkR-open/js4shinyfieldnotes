function prompt_name(idout){
  var nom = prompt("Enter a name for your scenario");

    return;
  }
  Shiny.setInputValue(idout, nom,  {priority: 'event'});
}

  var clientHeight = document.getElementById(what).clientWidth;
  Shiny.setInputValue('divheight', clientHeight);
});

var add_val_and_send = function(e, id, nom){
  com = "";
  $(e).data('value', $(e).data('value') + 1 );
  Shiny.setInputValue(id, {com : nom, val : $(e).data('value')}, {priority : "event"});
}

7.8 Update a custom Select Input

Shiny.addCustomMessageHandler(
  'update_select_menu', 
  function(e) {
    
    var select = $("#" + e.id_select);
    for (i = 0; i < e.vals.length; i++) { 
      var newopt = document.createElement( "option" );
      newopt.value = e.vals[i];
      newopt.innerText = e.vals[i];
      select.append(newopt);
    }
    
    var div = $("#" + e.id_div);
    
    if ( div.find("button").length === 0 ) {
      
    var newbtn = document.createElement( "button" );
    newbtn.setAttribute("type", "button");
    newbtn.setAttribute("onclick", "var x = $('#" + 
      e.id_select + "').val(); suppress_mesure(this, '"+ 
      e.id_out + "', x, '" + 
      e.id_div +"');");
    newbtn.innerText = "Valider";
    
    div.append(newbtn);
     
    var newbtn = document.createElement( "button" );
    newbtn.setAttribute("type", "button");
    newbtn.setAttribute("onclick", "$('#" + e.id_div +"').hide();");
    newbtn.innerText = "Annuler";
    
    div.append(newbtn) 
    }
    
  })
session$sendCustomMessage(
  "update_select_menu", 
  list(
    vals = local$vals, 
    id_div = ns("div_mesure"), 
    id_select = ns("select_mesure"), 
    id_out = ns("validate")
  )
)

7.10 Add specific elements

$( document ).ready(function() {
  Shiny.addCustomMessageHandler('show_year', function(arg) {
    $("#single > div.col-sm-3.well > div:nth-child(5)").hide()
    $("#single > div.col-sm-3.well > div:nth-child(7)").hide()
    $("#single > div.col-sm-3.well > div:nth-child(3)").show()
  })
  Shiny.addCustomMessageHandler('show_artist', function(arg) {
    $("#single > div.col-sm-3.well > div:nth-child(3)").hide()
    $("#single > div.col-sm-3.well > div:nth-child(7)").hide()
    $("#single > div.col-sm-3.well > div:nth-child(5)").show()
  })
  Shiny.addCustomMessageHandler('show_label', function(arg) {
    $("#single > div.col-sm-3.well > div:nth-child(3)").hide()
    $("#single > div.col-sm-3.well > div:nth-child(5)").hide()
    $("#single > div.col-sm-3.well > div:nth-child(7)").show()
  })
  
  Shiny.addCustomMessageHandler('success', function(content) {
    $("#alert-success").show();
  });
  
});
